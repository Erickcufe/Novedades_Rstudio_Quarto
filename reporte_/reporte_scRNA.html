untar(raw.path, exdir = file.path(tempdir(), "pbmc4k"))

## Calculando el tamaño de las librerias

lib.sf.pbmc <- librarySizeFactors(sce.pbmc) # <1>
ls.pbmc <- colSums(counts(sce.pbmc)) # <2>
plot(ls.pbmc, lib.sf.pbmc, log="xy", # <3>
